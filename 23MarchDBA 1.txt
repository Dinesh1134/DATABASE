-- block to search for employee

DECLARE
	v_name VARCHAR(20);
	emp_id employees.employee_id%TYPE:=190;
BEGIN
	SELECT first_name INTO v_name 
	FROM employees
	WHERE employee_id=emp_id;
	DBMS_OUTPUT.Put_Line('The employee is '||v_name);
END;
/


-- create a procedure to find employee by employee_id
CREATE OR REPLACE PROCEDURE proc_find(p_empid IN employees.employee_id%TYPE)
AS
	v_name VARCHAR(20);
BEGIN
	SELECT first_name INTO v_name
	FROM employees
	WHERE employee_id=p_empid;
	DBMS_OUTPUT.Put_Line('The employee is '||v_name);
END proc_find;
/

-- executing the procedure
exec proc_find(200)

-- add a new record into the teacher table 

CREATE OR REPLACE PROCEDURE proc_add(p_id IN INT,
					p_name IN VARCHAR,
					p_subject IN VARCHAR)
AS
BEGIN
		INSERT INTO teacher(id,name,subject) VALUES
		(p_id,p_name,p_subject);
END proc_add;
/


-- compile the block by
@insert

-- execute the block

exec proc_add(120,'Ramesh','C++')
